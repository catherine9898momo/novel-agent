# 小说生成系统 - 技术实现方案

## 技术栈选择

### 后端核心
- **语言**: TypeScript 5.x
- **运行时**: Node.js 18+
- **LLM SDK**: Anthropic SDK / OpenAI SDK
- **消息队列**: 基于文件系统的 JSONL 队列
- **任务调度**: 自研轻量级调度器
- **并发控制**: Worker Threads

### 数据存储
- **配置存储**: JSON 文件
- **内容存储**: Markdown 文件
- **状态管理**: SQLite（轻量级）
- **缓存**: 内存 + 文件系统

### 工具链
- **构建工具**: esbuild
- **测试框架**: Vitest
- **代码规范**: ESLint + Prettier
- **类型检查**: TypeScript strict mode

```typescript
// src/core/agent.ts

interface Message {
  role: 'user' | 'assistant' | 'system'
  content: string | ToolUse[]
}

interface ToolUse {
  type: 'tool_use'
  name: string
  input: Record<string, any>
}

interface ToolResult {
  type: 'tool_result'
  tool_use_id: string
  content: string
}

abstract class BaseAgent {
  protected messages: Message[] = []
  protected tools: ToolRegistry
  protected config: AgentConfig

  constructor(config: AgentConfig) {
    this.config = config
    this.tools = new ToolRegistry()
    this.registerTools()
  }

  // 子类实现具体工具注册
  protected abstract registerTools(): void

  // 核心 Agent Loop
  async run(task: Task): Promise<string> {
    this.messages.push({
      role: 'user',
      content: task.prompt
    })

    let iterations = 0
    const maxIterations = 50

    while (iterations < maxIterations) {
      const response = await this.callLLM()
      this.messages.push(response)

      // 检查是否有工具调用
      const toolCalls = this.extractToolCalls(response)
      if (toolCalls.length === 0) {
        // 没有工具调用，返回最终结果
        return response.content as string
      }

      // 执行所有工具调用
      for (const toolCall of toolCalls) {
        const result = await this.executeTool(toolCall)
        this.messages.push({
          role: 'user',
          content: result
        })
      }

      iterations++
    }

    throw new Error('Max iterations reached')
  }

  private async callLLM(): Promise<Message> {
    const response = await anthropic.messages.create({
      model: this.config.model || 'claude-sonnet-4-5',
      max_tokens: 4096,
      messages: this.messages,
      tools: this.tools.getDefinitions(),
      system: this.config.systemPrompt
    })

    return {
      role: 'assistant',
      content: response.content
    }
  }

  private async executeTool(toolCall: ToolUse): Promise<ToolResult> {
    const handler = this.tools.get(toolCall.name)
    if (!handler) {
      throw new Error(`Unknown tool: ${toolCall.name}`)
    }

    try {
      const result = await handler(toolCall.input)
      return {
        type: 'tool_result',
        tool_use_id: toolCall.id,
        content: JSON.stringify(result)
      }
    } catch (error) {
      return {
        type: 'tool_result',
        tool_use_id: toolCall.id,
        content: `Error: ${error.message}`
      }
    }
  }

  private extractToolCalls(response: Message): ToolUse[] {
    if (typeof response.content === 'string') {
      return []
    }
    return response.content.filter(c => c.type === 'tool_use')
  }
}
```

```typescript
// src/core/tool-registry.ts

interface ToolDefinition {
  name: string
  description: string
  input_schema: {
    type: 'object'
    properties: Record<string, any>
    required?: string[]
  }
}

type ToolHandler = (input: any) => Promise<any>

class ToolRegistry {
  private tools: Map<string, ToolHandler> = new Map()
  private definitions: ToolDefinition[] = []

  register(definition: ToolDefinition, handler: ToolHandler): void {
    this.tools.set(definition.name, handler)
    this.definitions.push(definition)
  }

  get(name: string): ToolHandler | undefined {
    return this.tools.get(name)
  }

  getDefinitions(): ToolDefinition[] {
    return this.definitions
  }
}
```

```typescript
    }
  }
```

```typescript
interface Message {
  id: string
  from: string
  to: string
  type: string
  payload: any
  timestamp: number
  request_id?: string
}

class MessageQueue {
  private inboxDir = '.team/inbox'
  private outboxDir = '.team/outbox'

  async send(to: string, message: any): Promise<void> {
    const msg: Message = {
      id: generateId(),
      from: 'lead',
      to,
      type: message.type,
      payload: message,
      timestamp: Date.now()
    }

    const inboxFile = path.join(this.inboxDir, `${to}.jsonl`)
    await appendFile(inboxFile, JSON.stringify(msg) + '\n')
  }

  async receive(agentId: string): Promise<Message[]> {
    const inboxFile = path.join(this.inboxDir, `${agentId}.jsonl`)

    if (!await exists(inboxFile)) {
      return []
    }

    const content = await readFile(inboxFile, 'utf-8')
    const messages = content
      .split('\n')
      .filter(line => line.trim())
      .map(line => JSON.parse(line))

    // 清空收件箱
    await writeFile(inboxFile, '')

    return messages
  }

  async reply(originalMessage: Message, response: any): Promise<void> {
    const msg: Message = {
      id: generateId(),
      from: originalMessage.to,
      to: originalMessage.from,
      type: `${originalMessage.type}_response`,
      payload: response,
      timestamp: Date.now(),
      request_id: originalMessage.id
    }

    const inboxFile = path.join(this.inboxDir, `${msg.to}.jsonl`)
    await appendFile(inboxFile, JSON.stringify(msg) + '\n')
  }
}
```

```typescript
// src/core/task-scheduler.ts

interface Task {
  id: string
  type: string
  status: 'pending' | 'assigned' | 'in_progress' | 'review' | 'completed'
  assignedTo?: string
  blockedBy: string[]
  data: any
  createdAt: number
  updatedAt: number
}

class TaskScheduler {
  private tasksDir = '.tasks'
  private db: Database // SQLite

  async addTask(task: Partial<Task>): Promise<Task> {
    const fullTask: Task = {
      id: task.id || generateId(),
      type: task.type!,
      status: 'pending',
      blockedBy: task.blockedBy || [],
      data: task.data || {},
      createdAt: Date.now(),
      updatedAt: Date.now()
    }

    await this.db.insert('tasks', fullTask)
    await this.saveTaskFile(fullTask)

    return fullTask
  }

  async getAvailableTasks(): Promise<Task[]> {
    // 获取未被阻塞的待处理任务
    const tasks = await this.db.query(`
      SELECT * FROM tasks
      WHERE status = 'pending'
      AND id NOT IN (
        SELECT task_id FROM task_blocks
        WHERE blocker_status != 'completed'
      )
    `)

    return tasks
  }

  async claimTask(taskId: string, agentId: string): Promise<boolean> {
    const task = await this.db.get('tasks', taskId)

    if (task.status !== 'pending') {
      return false
    }

    await this.db.update('tasks', taskId, {
      status: 'assigned',
      assignedTo: agentId,
      updatedAt: Date.now()
    })

    return true
  }

  async updateTaskStatus(
    taskId: string,
    status: Task['status']
  ): Promise<void> {
    await this.db.update('tasks', taskId, {
      status,
      updatedAt: Date.now()
    })

    // 如果任务完成，检查是否有被阻塞的任务可以释放
    if (status === 'completed') {
      await this.releaseBlockedTasks(taskId)
    }
  }

  private async releaseBlockedTasks(completedTaskId: string): Promise<void> {
    // 查找所有被此任务阻塞的任务
    const blockedTasks = await this.db.query(`
      SELECT * FROM tasks
      WHERE blockedBy LIKE '%${completedTaskId}%'
    `)

    for (const task of blockedTasks) {
      const blockers = task.blockedBy.filter(id => id !== completedTaskId)

      if (blockers.length === 0) {
        // 所有阻塞都已解除，任务可以执行
        await this.db.update('tasks', task.id, {
          blockedBy: [],
          status: 'pending'
        })
      }
    }
  }
}
```
